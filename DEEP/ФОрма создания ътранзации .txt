class NewTransactionWindow(ctk.CTkToplevel):
    def __init__(self, app_instance, *args, **kwargs):
        # ... остальной код инициализации ...
        
        # Фрейм для горизонтального размещения
        self.receipt_frame = ctk.CTkFrame(self, fg_color="transparent")
        self.receipt_frame.grid(row=1, column=3, sticky="we", padx=20, pady=5)
        self.receipt_frame.grid_columnconfigure(1, weight=1)
        
        # Кнопка выбора файла
        self.receipt_button = ctk.CTkButton(self.receipt_frame, text_color="black", text="Выбрать чек",
                                           width=100, command=self.select_receipt_file)
        self.receipt_button.grid(row=0, column=0, sticky="w", padx=(0, 10))
        
        # Метка с именем файла
        self.receipt_label = ctk.CTkLabel(self.receipt_frame, text_color="black", text="Чек не выбран",
                                         anchor="w")
        self.receipt_label.grid(row=0, column=1, sticky="we")
        
        # Остальные элементы сдвигаются вниз
        self.comment_entry.grid(row=2, column=3, sticky="swe", padx=20, pady=5)
        self.amount_entry.grid(row=3, column=3, sticky="nwe", padx=20, pady=5)

def select_receipt_file(self):
    file_path = filedialog.askopenfilename(
        title="Выберите файл чека",
        filetypes=[("Image files", "*.jpg *.jpeg *.png"), ("All files", "*.*")]
    )
    
    if file_path:
        try:
            # Читаем файл как бинарные данные
            with open(file_path, 'rb') as file:
                self.selected_receipt_data = file.read()
            
            # Показываем имя файла в интерфейсе (обрезаем если слишком длинное)
            file_name = os.path.basename(file_path)
            display_name = file_name if len(file_name) <= 20 else file_name[:17] + "..."
            self.receipt_label.configure(text=display_name)
            
        except Exception as e:
            CTkMessagebox.messagebox(title="Ошибка!", text=f"Не удалось загрузить файл: {str(e)}")
            self.selected_receipt_data = None
            self.receipt_label.configure(text="Чек не выбран")
